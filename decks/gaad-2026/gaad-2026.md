  function wrapInMark(node, step) {
    const mark = document.createElement('mark');
    mark.classList.add('error');
    mark.dataset.step = step;
    node.replaceWith(mark);
    mark.appendChild(node);
  }
  function updateMarks(slide) {
    const step = Number(slide.getAttribute('step')) || 0;
    for (const mark of slide.querySelectorAll('mark.error[data-step]')) {
      const markStep = Number(mark.dataset.step)
      mark.classList.toggle('revealed', step >= markStep);
      mark.classList.toggle('active', step === markStep);
    }
  }

    const slide = script.closest('rh-slide, rh-slide-sidebar');
    if (!slide) continue;
    for (const { selector, step, textContent } of JSON.parse(script.textContent)) {
      for (const el of slide.querySelectorAll(selector)) {
        if (!textContent) { wrapInMark(el, step); continue; }
        if (el.textContent.trim() === textContent) { wrapInMark(el, step); continue; }
        for (const node of el.childNodes) {
          if (node.nodeType === Node.TEXT_NODE && node.textContent.trim() === textContent) {
            const span = document.createElement('span');
            span.textContent = node.textContent;
            node.replaceWith(span);
            wrapInMark(span, step);
            break;
          }
        }
      }
    }
    updateMarks(slide);
  }

<abbr title="application programmer interfaces">**APIs**</abbr>,
**design tokens**, & **metadata** that are easily digestible by AI agents.

Not a new design system, a new way to **describe** one.

## <abbr title="agentic software development lifecycle">ASDL</abbr> horrors: hallucinations

<div class="grid-2">
  <div>
    <rh-icon class="center brand-red size-06"
             set="standard"
             loading="eager"
             icon="code"></rh-icon>

<abbr title="hypertext markup language">**HTML**</abbr>

Agents hallucinate **component APIs**,
leading to **incorrect** and **inaccessible** UIs:

- Invented attributes.
- Nonexistent slot names.
- Ignored <abbr title="accessibility">a11y</abbr> patterns.

  </div>
  <div>
    <rh-icon class="center brand-red size-06"
             set="ui"
             loading="eager"
             icon="ai-search"></rh-icon>

<abbr title="cascading stylesheets">**CSS**</abbr>

Models need _context_ to use **design tokens**,
and might not expend the effort to fetch docs from the web.

- Hallucinated token names
- Incorrect color values
- No semantic awareness

  </div>
</div>

**Semantic awareness**: the difference between "red-30" and "brand-red",
or "blue-20" and "interactive text".
{slot="presenter"}
